# column_variables.f90 you can do these changes or replace the whole column_variables.f90 file from Mg2 version of Kid
In this file around line ~45 where atmospheric variables are declared.
Add exactlythe lines below
!++ag
real(kind=wp) ::                          &
     exner_half(0:nz,0:nx+1)            & ! exner pressure
    ,pmb_half(0:nz,0:nx+1) = unset_real   ! pressure (mb) on interfaces
!--ag

# derived_fields.f90 you can do these changes or replace the whole derived_fields.f90 file from Mg2 version of Kid
modify the USE statement.
Current:
Use physconst, only : r_on_cp, g, p0, pi

Replace with:
Use physconst, only : r_on_cp, g, p0, pi, cp
